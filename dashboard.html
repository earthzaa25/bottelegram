ติดตั้ง Python และ Libraries
pip install requests pandas openpyxl
3
สร้างไฟล์ procurement_bot.py
import requests import pandas as pd from datetime import datetime BOT_TOKEN = "YOUR_BOT_TOKEN_HERE" CHAT_ID = "YOUR_GROUP_CHAT_ID" EXCEL_FILE = "รายการจัดซื้อจัดจ้าง กบ.ทหาร.xlsx" def send_daily_report(): df = pd.read_excel(EXCEL_FILE, sheet_name="ชีต1", header=None) data = df.iloc[2:].copy() data.columns = range(len(data.columns)) # ดึงงานที่มีข้อมูล rows = [] for _, r in data.iterrows(): no, name, budget = r[1], r[5], r[8] if pd.notna(no) and pd.notna(name) and str(no) not in ['nan','']: try: rows.append({'no': str(no), 'name': str(name), 'budget': float(budget) if pd.notna(budget) else 0}) except: pass total = len(rows) budget = sum(r['budget'] for r in rows) today = datetime.now().strftime("%d/%m/%Y %H:%M") msg = f""" 📊 *รายงานจัดซื้อจัดจ้าง กบ.ทหาร* 📅 ข้อมูล ณ {today} ━━━━━━━━━━━━━━━━━━━━ 📦 งานทั้งหมด: *{total} รายการ* 💰 วงเงินรวม: *{budget/1e6:,.1f} ล้านบาท* 🔴 *งานเร่งด่วน:* • จ้างสร้างเรือน้ำมัน ทร. — ค้าง 32 วัน • BEECHCRAFT 350i — รอ จก.กบ.ทหาร • Anti-Drone ทอ. — คาดเสนอ เม.ย.69 📎 ดูรายละเอียดครบถ้วนในไฟล์แนบ #กบทหาร #จัดซื้อจัดจ้าง """.strip() # ส่งข้อความ requests.post( f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage", json={"chat_id": CHAT_ID, "text": msg, "parse_mode": "Markdown"} ) # ส่งไฟล์ Excel แนบ with open(EXCEL_FILE, "rb") as f: requests.post( f"https://api.telegram.org/bot{BOT_TOKEN}/sendDocument", data={"chat_id": CHAT_ID, "caption": "📎 รายการจัดซื้อจัดจ้างล่าสุด"}, files={"document": f} ) print("✅ ส่งรายงานสำเร็จ") send_daily_report()
